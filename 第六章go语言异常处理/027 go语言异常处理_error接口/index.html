    error接口(常规错误)¶

    Go语言引入了一个关于错误处理的标准模式，即error接口，它是Go语言内建的接口类型，该接口的定义如下：(没有 try catch之类的机制)

    1
2
3
    type error interface {
    Error() string //一个方法,返回值是一个字符串
}

    Go语言的标准库代码包errors为用户提供如下方法：

     1
 2
 3
 4

    package errors

type errorString struct { 
    text string 
}

func New(text string) error { 
    return &errorString{text} 
} //封装了一个函数 ,最后需要调用的函数

func (e *errorString) Error() string { 
    return e.text //实现接口的方法
}

    另一个可以生成error类型值的方法是调用fmt包中的Errorf函数：

    1
2
3
4
5
6
    package fmt
import "errors"

func Errorf(format string, args ...interface{}) error {
    return errors.New(Sprintf(format, args...))
}

    "errors"
    "fmt"
)

func main() {
    var err1 error = errors.New("a normal err1")  //首先声明一个error接口变量
    fmt.Println(err1) //a normal err1

    var err2 error = fmt.Errorf("%s", "a normal err2") //使用fmt的函数实现
    fmt.Println(err2) //a normal err2
}

    import (
    "errors"
    "fmt"
)

func Divide(a, b float64) (result float64, err error) { //第一个括号是传递的参数,第二个是返回的参数
    if b == 0 {
        result = 0.0
        err = errors.New("runtime error: divide by zero")
        return
    }

    result = a / b
    err = nil
    return
}

func main() {
    r, err := Divide(10.0, 0)
    if err != nil {
        fmt.Println(err) //错误处理 runtime error: divide by zero
    } else {
        fmt.Println(r) // 使用返回值
    }
}

    package main

import "fmt"
import "errors"

func MyDiv(a, b int) (result int, err error) {

    err = nil  //首先定义err一开始是空,返回的参数已经声明了err是error类型,这里直接副职就可以了
    if b == 0 {
        err = errors.New("分母不能为0")  //新建了一种错误
    } else {
        result = a / b
    }

    return   //这里会自动的返回上面定义的返回变量,除非我们没有给err赋值
}

func main() {
    result, err := MyDiv(10, 0)
    if err != nil {
        fmt.Println("err = ", err)
    } else {
        fmt.Println("reslut = ", result)
    }

}
