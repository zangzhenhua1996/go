    go 变量定义

    Go 语言变量¶

    变量来源于数学，是计算机语言中能储存计算结果或能表示值抽象概念。

    变量可以通过变量名访问。

    Go 语言变量名由字母、数字、下划线组成，其中首个字符不能为数字。

    声明变量的一般形式是使用 var 关键字：

    变量声明¶

    第一种，指定变量类型，如果没有初始化，则变量默认为零值。

    零值就是变量没有做初始化时系统默认设置的值。 数值类型（包括complex64/128）为 0 布尔类型为 false 字符串为 ""（空字符串）

    第二种，根据值自行判定变量类型。

    第三种，省略 var, 注意 :=左侧如果没有声明新的变量，就产生编译错误，格式

    代码¶

    package main  //包名

import "fmt"  //导入打印的包

//不赋初值的变量声明,都是有默认值的
func variableZeroValue() {  //注意这一行的左花括号必须在这里
    var  a  int   //声明变量a是int型,其实更加符合阅读
    var  s  string  //在go语言中变量都是有定义的如果你给值那么整数就默认是0,字符串默认是空
    fmt.Println(a,s)  //使用这个Println并不会打印这个空的字符串
    fmt.Printf("%d %q\n",a,s) //使用格式化字符串打印这个不能显示的空字符串,这里使用的是Printf函数,并且呢使用的是%q进行格式化,会显示双引号
}

//赋初值的变量声明
func variableInitialValue() {
    var a,b int = 3,4 //go语言定义的变量必须用
    var s string = "abc"
    fmt.Println(a,b,s)
}

// 变量类型的自动推理,类似于js似的
func variableTypeDeduction() {
    var a,b = 3,4
    var s = "abc"
    var c,d = true,12
    fmt.Println(a,b,s,c,d)
}

// 使用 := 更加简略的声明变量
func variableshorter() {
    a,b,c,s := 3,4,true,"def"  //第一次出现的时候使用:=进行定义
    b=5  //重新赋值的时候就不能使用:= 了,一般我们定义变量都是使用 := 最佳的简洁
    fmt.Println(a,b,c,s)
}

//函数外的变量,作用域是包内,没有全局作用域之说,但是声明定义了就要使用
var aa =3
var ss ="kkk"
//bb:=true  在函数外是不能使用:=进行声明变量的
// 也可以使用下面的方式进行包内变量的定义
var (
    bb = 6
    cc =true
)

//主程序的入口
func main() {
    //打印
    fmt.Println("hello world")
    variableZeroValue()
    variableInitialValue()
    variableTypeDeduction()
    variableshorter()
    fmt.Println(aa,bb,cc,ss)
//输出结果
    /*
    hello world
    0
    0 ""
    3 4 abc
    3 4 abc true 12
    3 5 true def
    3 6 true kkk
    */
}
